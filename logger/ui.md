  function formatLogEntries(jsonArray) {
    return jsonArray.map((logEntry, id) => {
        const formattedTimestamp = new Date(logEntry.timestamp).toISOString().replace('T', ' ').split('.')[0];
        let actionDescription;

        switch (logEntry.type) {
            case "Termino.js Output:":
                actionDescription = `Output: ${logEntry.data}`;
                break;
            case "Termino.js Input:":
                actionDescription = `Input: ${logEntry.data}`;
                break;
            default:
                actionDescription = logEntry.data;
                break;
        }

        return `<p data-id="${id + 1}">${formattedTimestamp} - ${actionDescription}</p>`;
    }).join('');
}
